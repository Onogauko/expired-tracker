function switchView(v) { document.getElementById('view-dashboard') .classList.toggle('hidden', v !== 'dashboard'); document.getElementById('view-input') .classList.toggle('hidden', v !== 'input'); document.getElementById('view-history') .classList.toggle('hidden', v !== 'history'); document.getElementById('tab-dashboard').className = v === 'dashboard' ? "flex-1 py-3 text-xs font-black uppercase tracking-widest rounded-xl bg-blue-600 text-white shadow-md" : "flex-1 py-3 text-xs font-black uppercase tracking-widest rounded-xl text-slate-400"; document.getElementById('tab-input').className = v === 'input' ? "flex-1 py-3 text-xs font-black uppercase tracking-widest rounded-xl bg-blue-600 text-white shadow-md" : "flex-1 py-3 text-xs font-black uppercase tracking-widest rounded-xl text-slate-400"; document.getElementById('tab-history').className = v === 'history' ? "flex-1 py-3 text-xs font-black uppercase tracking-widest rounded-xl bg-blue-600 text-white shadow-md" : "flex-1 py-3 text-xs font-black uppercase tracking-widest rounded-xl text-slate-400"; } function updateDashboard(data) { const today = new Date().toISOString().split('T')[0]; let todayCount = 0; let expiredToday = 0; const picCount = {}; const chartMap = {}; data.forEach(item => { const inputDate = new Date(item.timestamp) .toISOString() .split('T')[0]; const expDate = new Date(item.expired) .toISOString() .split('T')[0]; if (inputDate === today) { todayCount++; } if (expDate === today) { expiredToday++; } if (!picCount[item.pic]) { picCount[item.pic] = 0; } picCount[item.pic]++; if (!chartMap[inputDate]) { chartMap[inputDate] = 0; } chartMap[inputDate]++; }); document.getElementById('statToday').innerText = todayCount; document.getElementById('statExpired').innerText = expiredToday; let topPic = "-"; let topCount = 0; for (const pic in picCount) { if (picCount[pic] > topCount) { topCount = picCount[pic]; topPic = `${pic} (${topCount})`; } } document.getElementById('topPic').innerText = topPic; const labels = Object.keys(chartMap).slice(-7); const values = Object.values(chartMap).slice(-7); const ctx = document .getElementById('weeklyChart') .getContext('2d'); if (window.myChart) { window.myChart.destroy(); } window.myChart = new Chart(ctx, { type: 'line', data: { labels: labels, datasets: [{ label: 'Input', data: values, tension: 0.4 }] }, options: { responsive: true, plugins: { legend: { display: false } } } }); } async function loadHistory() { const res = await fetch(API_URL + "?action=history"); const data = await res.json(); renderHistory(data); updateDashboard(data); }
